int romanToInt(char* s) {
  int num = 0;
  char Roman_numbers[7] = {'I', 'V', 'X', 'L', 'C', 'D', 'M'};

  for (int i = 0; i < strlen(s); i++) {
    if (s[i] == 'I') {
      if (s[i + 1] == 'V' || s[i + 1] == 'X') {
        num -= 1;
      } else {
        num++;
      }
    }

    if (s[i] == 'X') {
      if (s[i + 1] == 'L' || s[i + 1] == 'C') {
        num -= 10;
      } else {
        num += 10;
      }
    }

    if (s[i] == 'C') {
      if (s[i + 1] == 'D' || s[i + 1] == 'M') {
        num -= 100;
      } else {
        num += 100;
      }
    }
    if (s[i] == 'V') num += 5;
    if (s[i] == 'L') num += 50;
    if (s[i] == 'D') num += 500;
    if (s[i] == 'M') num += 1000;
  }

  return num;
}

func romanToInt(s string) int {
	m := map[byte]int{
		'I': 1,
		'V': 5,
		'X': 10,
		'L': 50,
		'C': 100,
		'D': 500,
		'M': 1000,
	}
	n := m[s[len(s) - 1]]
	for i := len(s) - 1; i > 0; i-- {
		preNum := m[s[i-1]]
		if m[s[i]] > preNum {
			n -= preNum
		}else {
			n += preNum
		}
	}
	return n
}
